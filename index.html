const canvas = document.getElementById('gameCanvas'); const ctx = canvas.getContext('2d'); // --- Postavke Platna --- canvas.width = 800; canvas.height = 400; // --- GAME DRŽAVA (STATE) --- let gameState = 'menu'; // menu, playing, dead let score = 0; let highScore = localStorage.getItem('jumperHighScore') || 0; // Čuvamo highscore u browseru // --- DEFINICIJA IGRAČA (BLOCK) --- let player = { x: 100, y: 300, width: 40, height: 40, color: 'red', // Početna boja dy: 0, // Brzina po Y osi gravity: 1.2, // Oštra gravitacija (bez air resistance) jumpForce: -20, onGround: false, // Svojstva za oči eyeBlinkTimer: 0, isBlinking: false, // Funkcija za skok jump: function() { if (this.onGround) { this.dy = this.jumpForce; this.onGround = false; } }, // Funkcija za ažuriranje fizike i očiju update: function() { // Fizika (čista gravitacija) this.dy += this.gravity; this.y += this.dy; // Provera poda if (this.y + this.height > canvas.height - 20) { this.y = canvas.height - 20 - this.height; this.dy = 0; this.onGround = true; } // Logika za treptanje očiju this.eyeBlinkTimer++; if (!this.isBlinking && this.eyeBlinkTimer > 150) { // Trepni svakih ~2.5 sekunde this.isBlinking = true; this.eyeBlinkTimer = 0; } if (this.isBlinking && this.eyeBlinkTimer > 10) { // Treptaj traje kratko this.isBlinking = false; this.eyeBlinkTimer = 0; } }, // Funkcija za crtanje kocke i očiju draw: function() { // Crtanje tela kocke ctx.fillStyle = this.color; ctx.fillRect(this.x, this.y, this.width, this.height); // Crtanje očiju (Bela osnova) ctx.fillStyle = 'white'; let eyeSize = 8; let eyeY = this.y + 10; let leftEyeX = this.x + 8; let rightEyeX = this.x + 24; if (this.isBlinking) { // Nacrtaj samo liniju ako trepće ctx.fillRect(leftEyeX, eyeY + 4, eyeSize, 2); ctx.fillRect(rightEyeX, eyeY + 4, eyeSize, 2); } else { // Normalne oči ctx.fillRect(leftEyeX, eyeY, eyeSize, eyeSize); ctx.fillRect(rightEyeX, eyeY, eyeSize, eyeSize); // Zenice ctx.fillStyle = 'black'; ctx.fillRect(leftEyeX + 2, eyeY + 2, 4, 4); ctx.fillRect(rightEyeX + 2, eyeY + 2, 4, 4); // LJUTE OČI (Ako je kocka Crvena) if (this.color === 'red') { ctx.strokeStyle = 'black'; ctx.lineWidth = 2; ctx.beginPath(); // Leva obrva ctx.moveTo(leftEyeX - 2, eyeY - 2); ctx.lineTo(leftEyeX + eyeSize + 2, eyeY + 3); // Desna obrva ctx.moveTo(rightEyeX - 2, eyeY + 3); ctx.lineTo(rightEyeX + eyeSize + 2, eyeY - 2); ctx.stroke(); } } } }; // --- PREPREKE (Samo na podu!) --- let obstacles = []; let obstacleTimer = 0; function spawnObstacle() { let size = 30; obstacles.push({ x: canvas.width, y: canvas.height - 20 - size, // Uvek na podu width: size, height: size, color: 'black' }); } // --- LOGIKA IGRE --- function resetGame() { score = 0; player.y = 300; player.dy = 0; obstacles = []; gameState = 'playing'; } function gameOver() { gameState = 'dead'; if (score > highScore) { highScore = score; localStorage.setItem('jumperHighScore', highScore); // Sačuvaj novi rekord } } // --- INPUTI (Tastatura/Klik) --- window.addEventListener('keydown', function(e) { if (e.code === 'Space') { if (gameState === 'playing') { player.jump(); } else if (gameState === 'menu' || gameState === 'dead') { resetGame(); } } }); // Klik na platno za promenu karaktera u meniju canvas.addEventListener('click', function(e) { if (gameState === 'menu' || gameState === 'dead') { const colors = ['red', 'yellow', 'green', 'blue']; let currentColorIndex = colors.indexOf(player.color); player.color = colors[(currentColorIndex + 1) % colors.length]; } }); // --- GLAVNA PETLJA (GAME LOOP) --- function loop() { ctx.clearRect(0, 0, canvas.width, canvas.height); // Crtanje poda ctx.fillStyle = '#333'; ctx.fillRect(0, canvas.height - 20, canvas.width, 20); if (gameState === 'menu') { // --- MENU SCREEN --- player.draw(); ctx.fillStyle = 'black'; ctx.font = '30px Arial'; ctx.textAlign = 'center'; ctx.fillText("JUMPER BLOCK", canvas.width / 2, canvas.height / 2 - 40); ctx.font = '20px Arial'; ctx.fillText("Pritisni SPACE za početak", canvas.width / 2, canvas.height / 2); ctx.fillText("Klikni na kocku da promeniš boju: " + player.color.toUpperCase(), canvas.width / 2, canvas.height / 2 + 30); ctx.fillText("High Score: " + highScore, canvas.width / 2, canvas.height / 2 + 70); } else if (gameState === 'playing') { // --- PLAYING SCREEN --- player.update(); player.draw(); // Upravljanje preprekama obstacleTimer++; if (obstacleTimer > 100) { // Spavnuj prepreku spawnObstacle(); obstacleTimer = 0; } for (let i = obstacles.length - 1; i >= 0; i--) { let o = obstacles[i]; o.x -= 8; // Brzina kretanja prepreka // Crtanje prepreke (Kvadrat na podu, bez spajkova) ctx.fillStyle = o.color; ctx.fillRect(o.x, o.y, o.width, o.height); // Provera kolizije if (player.x < o.x + o.width && player.x + player.width > o.x && player.y < o.y + o.height && player.y + player.height > o.y) { gameOver(); } // Ukloni prepreke koje su prošle if (o.x + o.width < 0) { obstacles.splice(i, 1); score++; // Povećaj skor kad preskočiš } } // Crtanje Skora ctx.fillStyle = 'black'; ctx.font = '24px Arial'; ctx.textAlign = 'left'; ctx.fillText("Score: " + score, 20, 40); } else if (gameState === 'dead') { // --- GAME OVER SCREEN --- player.draw(); for (let o of obstacles) { ctx.fillStyle = o.color; ctx.fillRect(o.x, o.y, o.width, o.height); } // Nacrtaj prepreke gde su stale ctx.fillStyle = 'rgba(0,0,0,0.5)'; ctx.fillRect(0,0, canvas.width, canvas.height); ctx.fillStyle = 'white'; ctx.font = '40px Arial'; ctx.textAlign = 'center'; ctx.fillText("GAME OVER", canvas.width / 2, canvas.height / 2 - 20); ctx.font = '24px Arial'; ctx.fillText("Tvoj Skor: " + score, canvas.width / 2, canvas.height / 2 + 20); ctx.fillText("Najbolji Skor: " + highScore, canvas.width / 2, canvas.height / 2 + 50); if (score >= highScore && score > 0) { ctx.fillStyle = 'yellow'; ctx.fillText("NOVI REKORD!", canvas.width / 2, canvas.height / 2 + 80); } else { ctx.fillText("Pokušaj da oboriš rekord!", canvas.width / 2, canvas.height / 2 + 80); } ctx.fillStyle = 'white'; ctx.font = '20px Arial'; ctx.fillText("Pritisni SPACE za ponovo", canvas.width / 2, canvas.height / 2 + 120); } requestAnimationFrame(loop); } loop();
